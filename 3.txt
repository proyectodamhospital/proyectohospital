const taskform = document.getElementById('task-from')

taskform.addEventListener('submit', (e)=>{
    e.preventDefault()
    const title = taskform['task-title']
    const descripcion = taskform['task-descripcion']


    saveTask(title.value, descripcion.value)
    taskform.reset()

})
